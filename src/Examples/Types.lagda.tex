%<*PropositionalEquality>
\begin{code}
data _≡_ {a} {A : Set a} (x : A) : A → Set a where
  refl : x ≡ x

trivial : 4 ≡ 4
trivial = refl
definitional : (2 + 2) ≡ 4
definitional = refl
\end{code}
%</PropositionalEquality>

%<*Cong>
\begin{code}
cong : ∀ {a} {A B : Set a} {m n : A} → (f : A → B) → m ≡ n → f m ≡ f n
cong f refl = refl


id-0 : (n : ℕ) → (n + 0) ≡ (0 + n)
id-0 zero = refl
id-0 (suc n) = cong suc (id-0 n)
\end{code}
%</Cong>
